Почувствуй разницу между attr() и prop()

attr()

// "check"
$('#check').attr('checked', true)
// "uncheck"
$('#check').attr('checked', false)

prop()

// check
$('#check').prop('checked', true)
// uncheck
$('#check').prop('checked', false)
// disable checkbox
$('#check').prop('disabled', true)
// enable checkbox
$('#check').prop('disabled', false)
