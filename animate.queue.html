// with queue
$('#box').animate({
    'left':'+=100'
}, 2000).animate({
    'top':'+=100'
}, 2000)
// without queue
$('#box').animate({
    'left':'-=100'
}, 2000).animate({
    'top':'-=100'
}, {queue:false, duration:2000})
// stop current
$('#box').stop()
// stop all animation
$('#box').stop(true)
// stop current animation, goto end, clear queue
$('#box').stop(true, true)
// stop current animation
$('#box').stop(false, true)
