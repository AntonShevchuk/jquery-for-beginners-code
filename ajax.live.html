$(function(){
    $('article').on('click', 'a', function(){
        var href = $(this).attr('href');
        $.ajax({
            url:href,
            dataType:'html',
            success:function(data){
                $('article').html(
                    $('article', data).contents()
                );
            }
        });
        return false;
    })
});
