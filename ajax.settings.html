// AJAX is easy
$.ajax({
  url: 'index.html',
  success: function(data) {
    let part = $(data).filter('main');
     $('article').append(part);
  }
})
// AJAX with Promise
$.ajax({
  url: 'index.html',
  success: function(data) {
    let part = $(data).filter('main');
    $('article').append(part);
  }
}).then(function() {
   $('article')
      .show()
      .fadeOut(2000);
})
