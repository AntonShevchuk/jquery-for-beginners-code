// AJAX is easy
$.ajax({
    url: 'index.html',
    success: function(data) {
        var part = $(data).filter('.jumbotron');
        $('article').append(part);
    }
})
// AJAX with Promise
$.ajax({
    url: 'index.html',
    success: function(data) {
        var part = $(data).filter('.jumbotron');
        $('article').append(part);
    }
}).then(function() {
     $('article').show().fadeOut(2000);
})
