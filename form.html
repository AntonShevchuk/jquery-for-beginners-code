Text input

$('input[type=text]').val('new text')
$('input[type=text]').val()

Text area

$('textarea').val('Some text')
$('textarea').val()

Selectbox

$('select').val('Option 2')
$('select').val()
$('select option:eq(2)').prop('selected', true)
$('select option:eq(2)').prop('selected', false)
$('select').append('<option>New Option</option>')
$('select').attr('size',
    $('select option').length
)
$('select option').remove()

Checboxes

$('input[type=checkbox]').prop('checked', true)
$('input[type=checkbox]').is(':checked')

Radiobuttons

$('input[type=radio][name=choose][value=2]')
    .prop('checked', true)
$('input[type=radio][name=choose][value=2]')
    .prop('checked', false)
$('input[type=radio][name=choose]:checked').val()
