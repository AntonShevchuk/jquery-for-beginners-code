// обработчики событий для first $('#first') .blur('counter') .focus('counter') .focusin('counter') .focusout('counter') // обработчики событий для last $('#last') .keydown('counter') .keypress('counter') .keyup('counter') // обработчики для textarea $('textarea') .change('counter') .select('counter') // обработчик для select $('select') .change('counter') // обработчик для checkbox $('input[name=checkbox]') .change('counter') // обработчик для radio buttons $('input[name=radio]') .change('counter') // обработчик для form $('form') .submit('counter')

Events:

  • 0change - изменение элемента
  • 0focus - фокус на элементе
  • 0blur - фокус ушел с элемента
  • 0focusin - фокус на элементе, событие предка
  • 0focusout - фокус ушел с элемента, событие предка
  • 0keydown - нажатие клавиши на клавиатуре
  • 0keypress - нажатие клавиши на клавиатуре keydown+keyup
  • 0keyup - отжатие клавиши на клавиатуре
  • 0select - выбор текста - актуален только для input[type=text] и textarea
  • 0submit - отправка формы
